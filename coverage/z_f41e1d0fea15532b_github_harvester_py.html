15class GithubHarvester: 

16 def __init__(self, id, logger, host="https://github.com", verbose=True): 

17 # Read Github API access token from config file. 

18 config = ConfigParser() 

19 config.read(os.path.join(os.path.abspath(os.path.dirname(__file__)), "../config/github.ini")) 

20 self.logger = logger 

21 self.id = id 

22 self.host = host 

23 self.verbose = verbose 

24 self.authenticate(config) 

25 self.data = {} # dictionary with all info 

26 fuji_server_dir = os.path.dirname(os.path.dirname(__file__)) # project_root 

27 software_file_path = os.path.join(fuji_server_dir, "data", "software_file.json") 

28 with open(software_file_path) as f: 

29 self.files_map = json.load(f)

31 def authenticate(self, config): 

32 """Runs every time a new harvesting request comes in, as the harvester is re-initialised every time. Picks a token (if available) and initialises the pyGithub handle. 

33 

34 Args: 

35 config (dict): parsed configuration dictionary 

36 """ 

37 token_file = config["ACCESS"]["token_file"] 

38 token_to_use = None 

39 if token_file != "": 

40 with open(token_file) as f: 

41 token_list = f.read().splitlines() 

42 # find a token with enough remaining requests, or the one with most remaining if none available 

43 fallback_max_token = None 

44 fallback_max_rate_limit = 0 

45 for token in token_list: 

46 try: 

47 rate_limit = Github(auth=Auth.Token(token)).get_rate_limit() 

48 if rate_limit.core.remaining >= 1000 and rate_limit.search.remaining >= 2: 

49 token_to_use = token 

50 break 

51 elif rate_limit.core.remaining > fallback_max_rate_limit: 

52 fallback_max_rate_limit = rate_limit.core.remaining 

53 fallback_max_token = token 

54 except: # ignore expired or invalid tokens 

55 pass 

56 if token_to_use is None: 

57 token_to_use = fallback_max_token 

58 else: 

59 token = config["ACCESS"]["token"] 

60 if token != "": 

61 token_to_use = token 

62 if token_to_use is not None: # found a token, one way or another 

63 auth = Auth.Token(token) 

64 if self.verbose: 

65 rate_limit = Github(auth=Auth.Token(token)).get_rate_limit() 

66 print( 

67 f"Authenticate using GitHub token ending on '{token[-4:]}'.\n Remaining core requests: {rate_limit.core.remaining}\n Remaining search requests: {rate_limit.search.remaining}" 

68 ) 

69 else: # empty token, so no authentication possible (rate limit will be much lower) 

70 auth = None 

71 self.logger.warning( 

72 "FRSM-09-A1 : Running in unauthenticated mode. Capabilities are limited." 

73 ) # TODO: would be better if it were a general warning! 

74 if self.host != "https://github.com": 

75 base_url = f"{self.host}/api/v3" 

76 self.handle = Github(auth=auth, base_url=base_url) 

77 else: 

78 self.handle = Github(auth=auth)

80 def harvest(self): 

81 tic = time.perf_counter() 

82 # check if it's a URL or repo ID 

83 # NOTE: this should probably be handled by IdentifierHelper, but I don't understand that module yet. 

84 if self.id.count("/") > 1: # URL 

85 _, self.username, self.repo_name = self.id.rsplit("/", 2) 

86 else: # repo ID 

87 self.username, self.repo_name = self.id.split("/") 

88 self.repo_id = "/".join([self.username, self.repo_name]) 

89 

90 # access repo via GitHub API 

91 try: 

92 repo = self.handle.get_repo(self.repo_id) 

93 except UnknownObjectException: 

94 print("Could not find repo.") 

95 self.logger.warning( 

96 "FRSM-09-A1 : Could not find repository on GitHub." 

97 ) # TODO: would be better if it were a general warning! 

98 return 

99 

100 # harvesting 

101 try: # LICENSE in metadata 

102 license_metadata = repo.get_license() 

103 self.data["license_path"] = license_metadata.path 

104 self.data["license"] = license_metadata.license.name 

105 except UnknownObjectException: 

106 pass 

107 

108 # identify source code (sample files in the main language used in the repo) 

109 repo_languages = repo.get_languages() 

110 if repo_languages != {}: 

111 self.data["languages"] = repo_languages 

112 main_source_code_language = repo.language 

113 if main_source_code_language is not None: 

114 self.data["main_language"] = main_source_code_language 

115 query = f' repo:{self.repo_id} language:"{main_source_code_language}"' # needs the space in front as every query needs a string to match on 

116 source_code_files = self.handle.search_code(query) 

117 # extract code of up to n=5 files 

118 n = min(5, source_code_files.totalCount) 

119 source_code_samples = [] 

120 for i in range(n): 

121 source_code_samples.append( 

122 { 

123 "path": source_code_files[i].path, 

124 "language": main_source_code_language, 

125 "content": source_code_files[i].decoded_content, 

126 } 

127 ) 

128 if len(source_code_samples) > 0: 

129 self.data["source_code_samples"] = source_code_samples 

130 

131 self.retrieve_all(repo) 

132 toc = time.perf_counter() 

133 if self.verbose: 

134 print(f"Harvesting took {toc - tic:.4f} seconds.") 

135 

136 def retrieve_all(self, repo): 

137 file_pattern = r"|".join([rf"(?P<{k}>{'|'.join(v['pattern'])})" for k, v in self.files_map.items()]) 

138 repo_contents = repo.get_contents("") 

139 while repo_contents: 

140 content_file = repo_contents.pop(0) 

141 if content_file.type == "dir": 

142 repo_contents.extend(repo.get_contents(content_file.path)) 

143 else: 

144 m = re.fullmatch(file_pattern, content_file.path) 

145 if m is not None and any(m.groupdict().values()): 

146 for k, v in m.groupdict().items(): 

147 if v is not None: 

148 if self.files_map[k]["parse"] == "full": 

149 file_entry = { 

150 "name": content_file.name, 

151 "path": content_file.path, 

152 "content": content_file.decoded_content, 

153 } 

154 elif self.files_map[k]["parse"] == "file_name": 

155 file_entry = {"name": content_file.name, "path": content_file.path} 

156 else: 

157 self.logger.warning( 

158 f"FRSM-09-A1 : Parsing strategy {self.files_map[k]['parse']} is currently not implemented. Choose one of 'full' or 'file_name' for files {k}. Defaulting to parsing strategy 'file_name'." 

159 ) 

160 file_entry = {"name": content_file.name, "path": content_file.path} 

161 try: 

162 self.data[k].append(file_entry) 

163 except KeyError: 

164 self.data[k] = [file_entry]
