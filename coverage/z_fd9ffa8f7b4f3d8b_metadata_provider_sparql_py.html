14 """A metadata provider class to get the metadata from a SPARQL query 

15 

16 ...

18 Methods 

19 ------- 

20 getMetadataStandards() 

21 Method will return the metadata standards in the namespaces 

22 getMetadata(queryString) 

23 Method that will return RDF graph and content type given the SPARQL query 

24 getNamespaces() 

25 Method to get namespaces 

26

28 

29 def getMetadataStandards(self): 

30 """Method will return the matadata standards in the namespaces 

31 

32 Returns 

33 ------- 

34 dict 

35 A dictionary of metadata standards. 

36 

37 """ 

38 

39 standards = {v: k for v, k in enumerate(self.namespaces)} 

40 return standards 

41 

42 def getMetadata(self, queryString): 

43 """Method will return RDF graph and namespace given the SPARQL query 

44 

45 Returns 

46 ------- 

47 rdflib.ConjunctiveGraph 

48 RDF graph from the result of SPARQL query 

49 str 

50 Content type of the result of SPARQL query 

51 """ 

52 

53 wrapper = SPARQLWrapper(self.endpoint) 

54 wrapper.setQuery(queryString) 

55 wrapper.setReturnFormat(RDFXML) 

56 

57 rdf_graph = None 

58 content_type = None 

59 try: 

60 response = wrapper.query() # application/rdf+xml 

61 content_type = response.info()["content-type"].split(";")[0] 

62 if "html" in content_type: 

63 self.logger.warning( 

64 f"{self.metric_id} : Looks like not a valid SPARQL endpoint, content type -: {content_type} " 

65 ) 

66 else: 

67 rdf_graph = response.convert() # rdflib.graph.ConjunctiveGraph 

68 # print(rdf_graph.serialize(format='xml')) 

69 # https://www.w3.org/TR/sparql11-protocol/#query-success 

70 # The response body of a successful query operation with a 2XX response is either: 

71 # a SPARQL Results Document in XML, JSON, or CSV/TSV format (for SPARQL Query forms SELECT and ASK); or 

72 # an RDF graph [RDF-CONCEPTS] serialized, for example, in the RDF/XML syntax [RDF-XML], or an equivalent RDF graph serialization, for SPARQL Query forms DESCRIBE and CONSTRUCT 

73 if isinstance(rdf_graph, rdflib.graph.Graph) and len(rdf_graph) > 0: 

74 self.logger.info( 

75 "{} : number of triples found in the graph, format -: {} of {}".format( 

76 self.metric_id, len(rdf_graph), content_type 

77 ) 

78 ) 

79 graph_text = rdf_graph.serialize(format="ttl") 

80 

81 for n in rdf_graph.namespaces(): 

82 self.namespaces.append(str(n[1])) 

83 self.getNamespacesfromIRIs(graph_text) 

84 else: 

85 self.logger.warning(f"{self.metric_id} : SPARQL query returns NO result.") 

86 except HTTPError as err1: 

87 self.logger.warning(f"{self.metric_id} : HTTPError -: {err1}") 

88 except SPARQLExceptions.EndPointNotFound as err2: 

89 self.logger.warning(f"{self.metric_id} : SPARQLExceptions -: {err2}") 

90 return rdf_graph, content_type 

91 

92 def getNamespaces(self): 

93 return self.namespaces
