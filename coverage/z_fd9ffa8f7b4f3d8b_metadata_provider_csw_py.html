11class OGCCSWMetadataProvider(MetadataProvider): 

12 """A metadata provider class that will provide metadata from OGCCSW""" 

13 

14 csw_namespaces = { 

15 "csw": "http://www.opengis.net/cat/csw/2.0.2", 

16 "ogc": "http://www.opengis.net/ogc", 

17 "ows": "http://www.opengis.net/ows", 

18 "gmd": "http://www.isotc211.org/2005/gmd", 

19 }

21 def getMetadata(self): 

22 # http://ws.pangaea.de/oai/provider?verb=GetRecord&metadataPrefix=oai_dc&identifier=oai:pangaea.de:doi:10.1594/PANGAEA.66871 

23 # The nature of a resource identifier is outside the scope of the OAI-PMH. 

24 # To facilitate access to the resource associated with harvested metadata, repositories should use an element in 

25 # #metadata records to establish a linkage between the record (and the identifier of its item) and the identifier 

26 # URL, URN, DOI, etc.) of the associated resource. 

27 # #The mandatory Dublin Core format provides the identifier element that should be used for this purpose 

28 return None

30 def getMetadataStandards(self): 

31 """Method to get the metadata schema from the OGCCSW namespaces 

32 

33 Returns 

34 ------- 

35 dict 

36 A dictionary of schemas in OGCCSW 

37 """ 

38 csw_endpoint = self.endpoint.split("?")[0] 

39 csw_listmetadata_url = csw_endpoint + "?service=CSW&request=GetCapabilities" 

40 requestHelper = RequestHelper(url=csw_listmetadata_url, logInst=self.logger) 

41 requestHelper.setAcceptType(AcceptTypes.xml) 

42 response_type, xml = requestHelper.content_negotiate(self.metric_id) 

43 schemas = {} 

44 if xml: 

45 try: 

46 root = etree.fromstring(requestHelper.response_content) 

47 metadata_nodes = root.xpath( 

48 '//ows:Parameter[@name="outputSchema"]/ows:Value', namespaces=OGCCSWMetadataProvider.csw_namespaces 

49 ) 

50 for node in metadata_nodes: 

51 if node.text: 

52 if node.text not in self.namespaces: 

53 self.namespaces.append(str(node.text)) 

54 schemas[str(node.text)] = str(node.text) 

55 except: 

56 self.logger.info(f"{self.metric_id} : Could not parse XML response retrieved from OGC CSW endpoint") 

57 

58 return schemas 

59 

60 def getNamespaces(self): 

61 return self.namespaces
